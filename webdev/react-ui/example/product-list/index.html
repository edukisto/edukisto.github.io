src/index.jsx
import React, { StrictMode } from 'react';
import { createRoot } from 'react-dom/client';
import { App } from './components/App.jsx';

window.addEventListener('load', () => {
  // Находим в DOM элемент с атрибутом id, равным my-app-root.
  const element = document.getElementById('my-app-root');

  if (!element) {
    throw new Error('Элемент не найден.');
  }

  // Определяем его как корневой элемент для отображения компонентов React.
  const root = createRoot(element);

  // Отображаем компонент App.
  root.render(
    <StrictMode>
      <App />
    </StrictMode>
  );
});

src/components/ProductList.jsx
import React, { useEffect, useState } from 'react';
import styles from './ProductList.module.css';
import { Button } from '@mui/material';
import Card from '@mui/material/Card';
import CardActions from '@mui/material/CardActions';
import CardContent from '@mui/material/CardContent';
import CardMedia from '@mui/material/CardMedia';

// Компонент «Витрина».
function ProductList() {
  // Объявляем переменную состояния products
  // и функцию­‑мутатор setProducts для изменения значения products.
  const [products, setProducts] = useState([]);

  async function getProducts() {

    // Запрашиваем перечень товаров...
    const response = await fetch('https://api.escuelajs.co/api/v1/products/', {
      headers: {
        // ...в формате строки JSON.
        'Accept': 'application/vnd.api+json',
      },
      method: 'GET',
    });

    // Преобразуем строку JSON в массив.
    const doc = await response.json();

    // Обновляем значение переменной products.
    // JSON:API требует хранить данные в поле data,
    // но escuelajs.co нарушает стандарт.
    setProducts(doc);
  }

  // Функция useEffect позволяет синхронизировать компонент с внешней системой
  // (в данном случае — с БД через RESTful API).
  // Обязательный аргумент useEffect — анонимная функция.
  useEffect(() => {
    // Асинхронная ф‑ция возвращает объект Promise.
    // Его метод catch позволяет обработать ошибки.
    getProducts().catch(console.error);
  }, []);

  // Создаём массив элементов HTML article, содержащих карточки товаров.
  // Вспомогательный атрибут key нужен React для эффективной перерисовки.
  // В ProductItem отправляем информацию о продукте и функцию удаления.
  const items = products.map((product) => {
    return (
      <article key={product.id}>
        <Card variant="outlined" sx={{ maxWidth: 345 }} sy={2}>
          <CardMedia
            sx={{ height: 140 }}
            image="https://mpt.ru/upload/head.jpg"
            title="green iguana"
          />

          <CardContent>
            <h2 className={styles.x}>{product.title}</h2>
          </CardContent>

          <CardActions>
            <Button variant="contained">Заказать</Button>
          </CardActions>
        </Card>
      </article>
    );
  });

  return (
    <>
      <h1>Продукты</h1>

      {items}
    </>
  );
}

export {
  ProductList,
};
