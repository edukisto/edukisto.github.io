Пример выполнения задания

Установка React​

  1. Создадим каталог проекта.

    mkdir react-shop

  2. Инициализируем программный пакет Node.js.

    npm init

    В каталоге появится файл package.json.

  3. Установим пакеты, от которых зависит наш пакет.

    npm install react react-dom react-scripts

    Установленные пакеты будут размещены в каталоге node_modules и зарегистрированы в файле package.json, в разделе dependencies.

  4. Создадим файловую структуру пакета.

    mkdir public
    mkdir src
    mkdir src/components
    echo '' > src/index.js
    echo '' > public/index.html
    echo '' > src/components/App.js

Код​

  1. Создадим шаблон веб­‑страницы.

    react-shop/public/index.html
    <!DOCTYPE html>
    <html lang="ru-luna1918">
    <head>
      <meta charset="UTF-8">
      <meta name="viewport"
            content="width=device-width, initial-scale=1.0">
      <title>Магазин</title>
    </head>
    <body>
      <noscript>Включите JavaScript.</noscript>
      <div id="my-app-root"></div>
    </body>
    </html>

  2. Создадим компонент Button.

    react-shop/src/components/Button/Button.jsx
    import React from 'react';

    // Компонент «Кнопка».
    function Button(props) {

      // Параметр props содержит атрибуты и вложенные элементы, переданные из JSX.
      // Например, из разметки <Button a="1" b="2">c</Button> в props попадёт
      // {
      //   "a": "1",
      //   "b": "2",
      //   "children": "c"
      // }

      // Сообщаем, что компонент Button при рендере станет элементом HTML button.
      // Все атрибуты и дочерние элементы компонента будут переданы в button.
      // Фигурные скобки в JSX — оператор вывода.
      // Многоточие означает разложение объекта на свойства.
      return (
        <button {...props}>{props.children}</button>
      );
    }

    // Экспортируем функцию/⁠компонент Button.
    // Теперь Button можно импортировать посредством import.
    export {
      Button,
    };

  3. Создадим компонент ProductItem.

    react-shop/src/components/ProductItem/ProductItem.jsx
    import { Button } from '../Button/Button.jsx';

    // Функция, обрабатывающая нажатие на кнопку.
    function handleClick(id, removeProduct) {
      if (confirm(`Удалить продукт ${id}?`)) {
        removeProduct(id);
        console.log(`Продукт с идентификатором ${id} удалён.`);
      }
    }

    // Компонент «Карточка товара».
    function ProductItem({ product, removeProduct }) {

      // В ProductItem, как и в Button, передаётся параметр props,
      // но здесь мы решили извлечь из props только некоторые поля, в т. ч. product.
      // Мы ожидаем, что компонент будет использоваться следующим образом:
      // <ProductItem product={{
      //   id: 123,
      //   name: 'Молоко',
      //   price: 100.99,
      // }} />

      // Сообщаем, что ProductItem при рендере ставновится фрагментом HTML,
      // содержащим заголовок, цену и кнопку для удаления товара из БД.
      return (
        <>
          <h1>{product.name}</h1>
          <p>Цена: {product.price} руб.</p>
          <Button onClick={() => {
            handleClick(product.id, removeProduct);
          }}>Удалить</Button>
        </>
      );
    }

    // Экспортируем функцию/⁠компонент ProductItem.
    // Теперь ProductItem можно импортировать посредством import.
    export {
      ProductItem,
    };

  4. Создадим компонент ProductList.

    react-shop/src/components/ProductList/ProductList.jsx
    import { useEffect, useState } from 'react';
    import { ProductItem } from '../ProductItem/ProductItem.jsx';

    // Компонент «Витрина».
    function ProductList() {
      // Объявляем переменную состояния products
      // и функцию­‑мутатор setProducts для изменения значения products.
      const [products, setProducts] = useState([]);

      // Асинхронная функция, запрашивающая перечень товаров.
      async function fetchProducts() {
        // Запрашиваем перечень товаров...
        const response = await fetch('http://127.0.0.1:8000/api/products/', {
          headers: {
            // ...в формате строки JSON.
            'Accept': 'application/vnd.api+json',
          },
          method: 'GET',
        });
        // Преобразуем строку JSON в массив.
        const json = await response.json();
        // Обновляем значение products.
        setProducts(json.data);
      }

      async function removeProduct(id) {
        // Запрашиваем удаление товара.
        await fetch(`http://127.0.0.1:8000/api/products/${id}`, {
          method: 'DELETE',
        });
        fetchProducts();
      }

      // Функция useEffect позволяет синхронизировать компонент с внешней системой
      // (в данном случае — с БД через JSON API на Laravel).
      // Обязательный аргумент useEffect — анонимная функция.
      useEffect(function () {
        // Обработка возможных ошибок.
        fetchProducts().catch(console.error);
      }, []);

      // Создаём массив элементов HTML article, содержащих карточки товаров.
      // Вспомогательный атрибут key нужен React для эффективной перерисовки.
      // В ProductItem отправляем информацию о продукте и функцию удаления.
      const items = products.map(function (product) {
        return (
          <article key={product.id}>
            <ProductItem product={product} removeProduct={removeProduct} />
          </article>
        );
      });

      return (
        <>
          {items}
        </>
      );
    }

    // Экспортируем функцию/⁠компонент ProductList.
    // Теперь ProductList можно импортировать посредством import.
    export {
      ProductList,
    };

  5. Создадим компонент App.

    react-shop/src/components/App/App.jsx
    import { ProductList } from '../ProductList/ProductList.jsx';

    function App() {
      // Возвращаем отрендеренный компонент ProductList.
      return (
        <ProductList />
      );
    }

    // Экспортируем функцию App.
    // Теперь её можно импортировать посредством import.
    export {
      App,
    };

  6. Создадим главный модуль веб­‑приложения.

    react-shop/src/index.jsx
    import { App } from '../src/components/App/App.jsx';
    import { createRoot } from 'react-dom/client';

    // @ts-check

    window.addEventListener('load', function () {
      // Найти на странице элемент с атрибутом id, равным my-app-root.
      const container = document.getElementById('my-app-root');

      // Задаём корневой элемент для отображения компонентов React.
      const root = createRoot(container);

      // Отображаем компонент App.
      root.render(<App />);
    });

  7. Запустим отладочный сервер.

    npx react-scripts start
    Клиенту нужен сервер

    Не забудьте в другом эмуляторе терминала запустить из проект на Laravel, который, нужно полагать, хранится в соседнем каталоге:

    cd путь/к/каталогу/с/проектом/на/Laravel/
    php artisan serve

  8. Откроем в браузере веб­‑страницу, URL которой выведен в консоль.
