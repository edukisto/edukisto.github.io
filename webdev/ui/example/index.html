Пример выполнения задания

Пользовательский интерфейс​

  1. Создадим каталог shop-ui и перейдём в него.

    mkdir shop-ui && cd shop-ui

  2. Откроем эмулятор терминала.

  3. Запустим отладочный веб­‑сервер:

    • PHP
    • Python
    php -S 127.0.0.2:8080
    python -m http.server --bind=127.0.0.2 8080

  4. Создадим файл index.html.

    index.html
    <!DOCTYPE html>
    <html lang="ru-luna1918">
    <head>
      <meta charset="UTF-8">
      <meta name="viewport"
            content="width=device-width, initial-scale=1.0">
      <title>Магазин</title>
      <!-- Подключаем файл со сценарием JavaScript. -->
      <script src="index.js"
              type="module"></script>
    </head>
    <body>
      <article>
        <h1>Товары</h1>
        <section>
          <h2>Добавление записи</h2>

          <!-- Форма для создания записи о товаре. -->
          <!-- Значения атрибутов name равны именам столбцов в таблице БД. -->
          <form action="https://fakestoreapi.com/products/"
                id="product-add-form"
                method="post">
            <label>
              Наименование:
              <input name="name"
                     required
                     value="Молоко">
            </label>
            <label>
              Цена:
              <input min="0.01"
                     name="price"
                     required
                     step="0.01"
                     type="number"
                     value="109.99">
            </label>

            <!-- Для удобства доступа из JavaScript даём кнопке идентификатор. -->
            <button type="submit">Сохранить</button>
          </form>
        </section>
        <section>
          <h2>Витрина</h2>
          <!-- Для удобства доступа из JavaScript даём контейнеру идентификатор. -->
          <div id="catalog"></div>
        </section>
      </article>
    </body>

    </html>

  5. Создадим файл components/catalog.js.

    components/catalog.js
    import { ProductCard } from './product-card.js';

    async function getProducts() {
      // URL ресурса API, который принимает запросы на выборку всех записей из БД.
      const uri = 'https://fakestoreapi.com/products';

      // Подаём HTTP-запрос.
      const response = await fetch(uri);

      // Парсим JSON, пришедший в теле ответа сервера.
      const products = await response.json();

      return products;
    }

    function Catalog(container) {
      // Запрашиваем список продуктов.
      const promise = getProducts();

      promise.then((products) => {
        // Создаём в ОЗУ контейнер для элементов HTML.
        const articles = document.createDocumentFragment();

        // Перебираем записи о товарах.
        for (const product of products) {
          const article = ProductCard(product);
          articles.appendChild(article);
        }

        // Отображаем элементы.
        container.appendChild(articles);
      });
    }

    export {
      Catalog
    };

  6. Создадим файл components/product-card.js.

    components/product-card.js
    import { destroyProduct } from './destroy-product.js';

    function ProductCard(product) {
      // Создаём контейнер для карточки товара.
      const article = document.createElement('article');

      // Даём ему идентификатор.
      article.id = `product-${product.id}`;

      // Создаём заголовок.
      const h1 = document.createElement('h1');
      h1.textContent = product.title;

      // Создаём кнопку удаления.
      const button = document.createElement('button');
      button.textContent = 'Удалить';
      // Обработчик щелчка мыши по кнопке.
      button.addEventListener('click', () => {
        destroyProduct(product);
      });

      article.appendChild(h1);
      article.appendChild(button);

      return article;
    }

    export {
      ProductCard
    };

  7. Создадим файл components/destroy-product.js.

    components/destroy-product.js
    async function destroyProduct(product) {
      // Формируем URI.
      const uri = `https://fakestoreapi.com/products/${product.id}`

      // Подаём на URI запрос HTTP методом DELETE.
      const response = await fetch(uri, { method: 'DELETE' });

      // Если статус ответа в диапазоне 200–299, то...
      if (response.ok) {
        // Выводим отладочное сообщение в консоль.
        console.log(`Продукт ${product.id} удалён.`);

        // Находим карточку товара на странице и удаляем её.
        const article = document.getElementById(`product-${product.id}`);
        article.parentElement.removeChild(article);
      } else {
        alert(`Ошибка удаления продукта ${product.id}!`);
      }
    }

    export {
      destroyProduct
    };

  8. Создадим файл components/product-add-form.js.

    components/product-add-form.js
    async function onSubmit(event) {
      // Отмена стандартного поведения — отправки формы с переходом.
      event.preventDefault();

      // event.target — это ссылка на форму, которую предъявил пользователь.

      // URI ресурса API, который принимает запросы на добавление записей в БД.
      const uri = event.target.action;

      // Собираем данные с формы.
      const data = new FormData(event.target);

      const json = JSON.stringify(Object.fromEntries(data.entries()));

      // Подаём на URI запрос HTTP методом POST.
      const response = await fetch(uri, {
        // Кодируем в JSON данные, взятые из формы.
        body: json,
        headers: {
          // Сообщаем серверу, что ожидаем ответ в формате JSON.
          'Accept': 'application/json',

          // Сообщаем серверу, что отправляем ему данные в формате JSON.
          'Content-Type': 'application/json',
        },
        method: 'POST',
      });

      // Если статус ответа в диапазоне 200–299, то...
      if (response.ok) {
        console.log(await response.json());

        alert('Запись добавлена');
      } else {
        alert('Ошибка!');
      }
    }

    function ProductAddForm(form) {
      form.addEventListener('submit', onSubmit);
    }

    export {
      ProductAddForm
    };

  9. Создадим файл index.js.

    index.js
    import { Catalog } from './components/catalog.js';
    import { ProductAddForm } from './components/product-add-form.js';

    window.addEventListener('load', () => {
      const catalog = new Catalog(document.getElementById('catalog'));
      const controller = new ProductAddForm(document.getElementById('product-add-form'));
    });

  10. Откроем в браузере веб­‑страницу http://127.0.0.2:8080/.
