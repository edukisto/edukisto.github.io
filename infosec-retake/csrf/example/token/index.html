Токен

Секретный ключ​

  1. Многим алгоритмам смягчения потребуется секретный ключ на стороне сервера, поэтому создадим файл generate_key.php, содержащий следующий сценарий:

    Файл utils/generate_key.php
    <?php

    // Имя файла с секретным ключом.
    $filename = '.key.txt';

    try {
        // Если файл уже существует...
        if (file_exists($filename)) {
            throw new Exception("Ключ уже записан в файл $filename.");
        }

        // Создаём секретный ключ ⁠— 256­‑битное значение, выданное КСГПСЧ.
        // Кодируем секретный ключ в base64.
        $key = base64_encode(random_bytes(32));

        // Если не удалось записать ключ в файл...
        if (false === file_put_contents($filename, $key, LOCK_EX)) {
            throw new Exception("Не удалось записать ключ в файл $filename.");
        }

        echo "Ключ записан в файл $filename.";
    } catch (Exception $exception) {
        echo $exception->getMessage();
    }

  2. Вызовем этот сценарий:

    php utils/generate_key.php

    В корневом каталоге должен появиться файл .key.txt с секретным ключом.

Смягчение на основе токенов​

Простейший токен​

Файл utils/basiccsrftoken/manager.php
<?php

declare(strict_types=1);

namespace BasicCsrfToken;

class Manager
{
    // Метод генерации CSRF­‑токена.
    function make(): string
    {
        // Посредством КСГПСЧ генерируем 256­‑битное значение.
        // Кодируем его в base64.
        return base64_encode(random_bytes(32));
    }

    // Метод проверки CSRF­‑токена.
    function check(string $token): bool
    {
        // Декодируем сообщение из base64.
        $message = base64_decode($token, true);
        if (false === $message) {
            return false;
        }
        // Проверяем хотя бы длину сообщения.
        return strlen($message) === 32;
    }
}

Пример генерации токена:

$manager = new Manager();
$_SESSION['csrf_token'] = $manager->make();

Пример проверки:

$manager = new Manager();
if ($manager->check(...)) {
    // ...
}

Шаблон токена, основанного на шифровании​

Пример реализации шаблона на PHP 7.1 и OpenSSL 1.0.1.

Файл utils/encryptionbasedcsrftoken/manager.php
<?php

declare(strict_types=1);

namespace EncryptionBasedCsrfToken;

class Manager
{
    private $cryptographer;

    public function __construct($cryptographer)
    {
        $this->cryptographer = $cryptographer;
    }

    // Метод генерации CSRF­‑токена.
    function make(int $expires, string $key): string
    {
        $encryptedMessage = $this->cryptographer->encrypt(
            // Конкатенируем ID сеанса с меткой времени,
            // используя «|» как разделитель.
            session_id() . '|' . $expires,

            // Указываем секретный ключ.
            $key
        );

        // Кодируем кортеж в base64 (английские буквы, цифры, «+», «/», «=»).
        return base64_encode("$encryptedMessage|$expires");
    }

    // Метод проверки CSRF­‑токена.
    function check(string $token, string $key): bool
    {
        // Декодируем нешифрованный кортеж из base64.
        $unencTuple = base64_decode($token);

        // Извлекаем шифрованный кортеж и нешифрованную метку времени.
        list($encTuple, $unenc_expires) = explode('|', $unencTuple);

        // Дешифруем кортеж.
        $decTuple = $this->cryptographer->decrypt($encTuple, $key);

        // Извлекаем из дешифрованного кортежа ID сеанса и метку времени.
        list($dec_session_id, $dec_expires) = explode('|', $decTuple);

        // Если метки времени не совпадают или сеанс окончен,
        var_dump($unenc_expires, $dec_expires, time());
        if (($unenc_expires !== $dec_expires) || (time() > $dec_expires)) {
            // то признаём токен недействительным.
            return false;
        }

        // Сравниваем дешифрованный ID сеанса с текущим.
        return $dec_session_id === session_id();
    }
}
Файл utils/encryptionbasedcsrftoken/aes256gcm.php
<?php

declare(strict_types=1);

namespace EncryptionBasedCsrfToken;

class AES256GCM
{
    private $aad;
    private $iv;
    private $ivLength;
    private $method;
    private $options;
    private $tag;
    private $tagLength;

    public function __construct(
        int $options = OPENSSL_RAW_DATA,
        string $iv = '',
        string &$tag = '',
        string $aad = ''
    ) {
        $this->aad = $aad;
        $this->method = 'aes-256-gcm';
        $this->ivLength = openssl_cipher_iv_length($this->method);
        $this->iv = openssl_random_pseudo_bytes($this->ivLength);
        $this->options = $options;
        $this->tag = $tag;
        $this->tagLength = 16;
    }

    public function decrypt(string $data, string $key): string
    {
        $data = base64_decode($data);
        $iv = substr($data, 0, $this->ivLength);
        $tag = substr($data, $this->ivLength, $this->tagLength);
        $message = substr($data, $this->ivLength + $this->tagLength);
        return openssl_decrypt(
            $message,
            $this->method,
            $key,
            $this->options,
            $iv,
            $tag,
            $this->aad
        );
    }

    public function encrypt(string $message, string $key): string
    {
        $encryptedMessage = openssl_encrypt(
            $message,
            $this->method,
            $key,
            $this->options,
            $this->iv,
            $this->tag,
            $this->aad,
            $this->tagLength
        );
        return base64_encode($this->iv . $this->tag . $encryptedMessage);
    }
}

Пример генерации токена:

// Срок действия сессии.
$lifetime = 60 * 60 * 24;
// Считываем секретный ключ.
$key = base64_decode(trim(file_get_contents('.key.txt')));
$manager = new Manager(new AES256GCM());
$expires = time() + $lifetime;
// Генерируем CSRF­‑токен.
$_SESSION['csrf_token'] = $manager->make($expires, $key);

Пример проверки:

$key = base64_decode(trim(file_get_contents('.key.txt')));
$manager = new Manager(new AES256GCM());
if ($manager->check($token, $key)) {
    // ...
}

Шаблон токена, основанного на HMAC​

Файл utils/hmacbasedcsrftoken/manager.php
<?php

declare(strict_types=1);

namespace HmacBasedCsrfToken;

class Manager
{
    // Метод генерации CSRF­‑токена.
    function make(int $expires, string $key): string
    {
        // Вывод перечня хеш­‑функций, подходящих для создания HMAC.
        // print_r(hash_hmac_algos());

        $hmac = hash_hmac(
            // Указываем желаемый алгоритм.
            'sha3-256',

            // Кодируем ID сеанса в base64 (английские буквы, цифры, «+», «/», «=»)
            // и конкатенируем с меткой времени, используя «|» как разделитель.
            base64_encode(session_id()) . '|' . $expires,

            // Указываем секретный ключ.
            $key
        );
        return base64_encode("$hmac|$expires");
    }

    // Метод проверки CSRF­‑токена.
    function check(string $token, string $key): bool
    {
        // Декодируем сообщение из base64.
        $message = base64_decode($token);

        // Извлекаем из сообщения метку времени.
        list(, $expires) = explode('|', $message);

        // Если сеанс окончен,
        if (time() > $expires) {
            // то признаём токен недействительным.
            return false;
        }

        // Сравниваем токен с эталоном.
        return $token === $this->make((int) $expires, $key);
    }
}

Примеры генерации и проверки токена такие же, как в предыдущем случае, но менеджеру не нужен внешний алгоритм шифрования.

$manager = new Manager(new AES256GCM());
$manager = new Manager();
